# Achates iOS App Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a native SwiftUI iOS app and supporting server protocol for multi-agent chat with device services.

**Architecture:** New `/ws/v2` endpoint with frame-based RPC+event protocol, handled by `MobileTransport` (independent of Gateway). iOS app uses `@Observable` state management with `URLSessionWebSocketTask`. Server and iOS are developed in parallel — server first, then iOS.

**Tech Stack:** .NET 10 (server), Swift 6 / SwiftUI (iOS), WebSocket (JSON frames), APNs (push)

```csharp
// src/Achates.Server/Mobile/MobileSessionStore.cs
using System.Text.Json;
using Achates.Agent.Messages;

namespace Achates.Server.Mobile;

public sealed class MobileSessionStore(string basePath)
{
    private static readonly JsonSerializerOptions JsonOptions = new()
    {
        PropertyNamingPolicy = JsonNamingPolicy.SnakeCaseLower,
        WriteIndented = true,
    };

    public async Task<MobileSession?> LoadAsync(string agentName, string peerId, string sessionId, CancellationToken ct = default)
    {
        var path = GetPath(agentName, peerId, sessionId);
        if (!File.Exists(path)) return null;
        await using var stream = File.OpenRead(path);
        return await JsonSerializer.DeserializeAsync<MobileSession>(stream, JsonOptions, ct);
    }

    public async Task SaveAsync(string agentName, string peerId, MobileSession session, CancellationToken ct = default)
    {
        var path = GetPath(agentName, peerId, session.Id);
        Directory.CreateDirectory(Path.GetDirectoryName(path)!);
        session.Updated = DateTimeOffset.UtcNow;
        await using var stream = File.Create(path);
        await JsonSerializer.SerializeAsync(stream, session, JsonOptions, ct);
    }

    public Task DeleteAsync(string agentName, string peerId, string sessionId, CancellationToken ct = default)
    {
        var path = GetPath(agentName, peerId, sessionId);
        if (File.Exists(path)) File.Delete(path);
        return Task.CompletedTask;
    }

    public async Task<IReadOnlyList<MobileSessionInfo>> ListAsync(string agentName, string peerId, CancellationToken ct = default)
    {
        var dir = GetDirectory(agentName, peerId);
        if (!Directory.Exists(dir)) return [];

        var files = Directory.GetFiles(dir, "*.json");
        var results = new List<MobileSessionInfo>(files.Length);

        foreach (var file in files)
        {
            var session = await LoadAsync(agentName, peerId, Path.GetFileNameWithoutExtension(file), ct);
            if (session is null) continue;

            var lastUserMessage = session.Messages.OfType<UserMessage>().LastOrDefault();
            results.Add(new MobileSessionInfo(
                session.Id,
                session.Title,
                session.Created,
                session.Updated,
                session.Messages.Count,
                lastUserMessage?.Text));
        }

        return results.OrderByDescending(s => s.Updated).ToList();
    }

    public async Task UpdateMetadataAsync(string agentName, string peerId, string sessionId, string title, CancellationToken ct = default)
    {
        var session = await LoadAsync(agentName, peerId, sessionId, ct);
        if (session is null) return;
        session.Title = title;
        await SaveAsync(agentName, peerId, session, ct);
    }

    private string GetPath(string agentName, string peerId, string sessionId)
        => Path.Combine(basePath, "agents", agentName, "sessions", "mobile", peerId, $"{sessionId}.json");

    private string GetDirectory(string agentName, string peerId)
        => Path.Combine(basePath, "agents", agentName, "sessions", "mobile", peerId);
}
```

```csharp
// src/Achates.Server/Mobile/Frames.cs
using System.Text.Json;
using System.Text.Json.Serialization;

namespace Achates.Server.Mobile;

public abstract record Frame
{
    [JsonPropertyName("type")]
    public abstract string Type { get; }
}

public sealed record RequestFrame : Frame
{
    public override string Type => "req";
    public required string Id { get; init; }
    public required string Method { get; init; }
    public JsonElement Params { get; init; } = default;
}

public sealed record ResponseFrame : Frame
{
    public override string Type => "res";
    public required string Id { get; init; }
    public required bool Ok { get; init; }
    public JsonElement? Payload { get; init; }
    public FrameError? Error { get; init; }

    public static ResponseFrame Success(string id, JsonElement payload) => new()
    {
        Id = id, Ok = true, Payload = payload,
    };

    public static ResponseFrame Failure(string id, string code, string message) => new()
    {
        Id = id, Ok = false, Error = new FrameError { Code = code, Message = message },
    };
}

public sealed record FrameError
{
    public required string Code { get; init; }
    public required string Message { get; init; }
}

public sealed record EventFrame : Frame
{
    public override string Type => "evt";
    public required string Event { get; init; }
    public JsonElement Payload { get; init; } = default;
    public required long Seq { get; init; }
}

public static class FrameParser
{
    private static readonly JsonSerializerOptions Options = new()
    {
        PropertyNamingPolicy = JsonNamingPolicy.SnakeCaseLower,
    };

    public static Frame Parse(string json)
    {
        using var doc = JsonDocument.Parse(json);
        var type = doc.RootElement.GetProperty("type").GetString();
        return type switch
        {
            "req" => JsonSerializer.Deserialize<RequestFrame>(json, Options)!,
            "res" => JsonSerializer.Deserialize<ResponseFrame>(json, Options)!,
            "evt" => JsonSerializer.Deserialize<EventFrame>(json, Options)!,
            _ => throw new JsonException($"Unknown frame type: {type}"),
        };
    }
}
```

```csharp
// src/Achates.Server/Mobile/MobileConnection.cs
using System.Collections.Concurrent;
using System.Net.WebSockets;
using System.Text;
using System.Text.Json;
using Achates.Agent;

namespace Achates.Server.Mobile;

/// <summary>
/// Represents a single active mobile WebSocket connection.
/// Manages RPC correlation, event sequencing, and agent runtime instances.
/// </summary>
public sealed class MobileConnection(WebSocket socket, string peerId) : IAsyncDisposable
{
    private static readonly JsonSerializerOptions JsonOptions = new()
    {
        PropertyNamingPolicy = JsonNamingPolicy.SnakeCaseLower,
        DefaultIgnoreCondition = System.Text.Json.Serialization.JsonIgnoreCondition.WhenWritingNull,
    };

    private readonly ConcurrentDictionary<string, TaskCompletionSource<ResponseFrame>> _pendingRequests = new();
    private readonly ConcurrentDictionary<string, AgentRuntime> _runtimes = new();
    private long _seq;

    public string PeerId => peerId;
    public WebSocket Socket => socket;
    public string? CurrentAgent { get; set; }
    public string? CurrentSessionId { get; set; }
    public HashSet<string> Capabilities { get; } = [];
    public string? DeviceToken { get; set; }

    public AgentRuntime? GetRuntime(string agentName)
        => _runtimes.GetValueOrDefault(agentName);

    public void SetRuntime(string agentName, AgentRuntime runtime)
        => _runtimes[agentName] = runtime;

    public void RemoveRuntime(string agentName)
        => _runtimes.TryRemove(agentName, out _);

    public long NextSeq() => Interlocked.Increment(ref _seq);

    public async Task SendFrameAsync(Frame frame, CancellationToken ct = default)
    {
        if (socket.State != WebSocketState.Open) return;
        var json = JsonSerializer.Serialize(frame, frame.GetType(), JsonOptions);
        var bytes = Encoding.UTF8.GetBytes(json);
        await socket.SendAsync(bytes, WebSocketMessageType.Text, true, ct);
    }

    public async Task SendEventAsync(string eventName, JsonElement payload, CancellationToken ct = default)
    {
        var frame = new EventFrame
        {
            Event = eventName,
            Payload = payload,
            Seq = NextSeq(),
        };
        await SendFrameAsync(frame, ct);
    }

    public async Task<ResponseFrame?> SendRequestAsync(string method, JsonElement? parameters = null, TimeSpan? timeout = null, CancellationToken ct = default)
    {
        var id = Guid.NewGuid().ToString("N");
        var tcs = new TaskCompletionSource<ResponseFrame>(TaskCreationOptions.RunContinuationsAsynchronously);
        _pendingRequests[id] = tcs;

        try
        {
            var frame = new RequestFrame
            {
                Id = id,
                Method = method,
                Params = parameters ?? JsonSerializer.SerializeToElement(new { }),
            };
            await SendFrameAsync(frame, ct);

            using var timeoutCts = new CancellationTokenSource(timeout ?? TimeSpan.FromSeconds(30));
            using var linked = CancellationTokenSource.CreateLinkedTokenSource(ct, timeoutCts.Token);
            linked.Token.Register(() => tcs.TrySetCanceled());

            return await tcs.Task;
        }
        finally
        {
            _pendingRequests.TryRemove(id, out _);
        }
    }

    public bool TryCompleteRequest(string id, ResponseFrame response)
    {
        if (_pendingRequests.TryRemove(id, out var tcs))
        {
            tcs.TrySetResult(response);
            return true;
        }
        return false;
    }

    public async ValueTask DisposeAsync()
    {
        foreach (var tcs in _pendingRequests.Values)
            tcs.TrySetCanceled();
        _pendingRequests.Clear();

        foreach (var runtime in _runtimes.Values)
        {
            runtime.Abort();
            await runtime.WaitForIdleAsync();
        }
        _runtimes.Clear();

        if (socket.State is WebSocketState.Open or WebSocketState.CloseReceived)
        {
            try { await socket.CloseAsync(WebSocketCloseStatus.NormalClosure, null, CancellationToken.None); }
            catch { /* best effort */ }
        }
    }
}
```

```csharp
// src/Achates.Server/Mobile/MobileTransport.cs
using System.Net.WebSockets;
using System.Text;
using System.Text.Json;
using Achates.Agent;
using Achates.Agent.Events;
using Achates.Agent.Messages;
using Achates.Providers.Completions.Events;
using Microsoft.Extensions.Logging;

namespace Achates.Server.Mobile;

/// <summary>
/// Handles mobile WebSocket connections at /ws/v2.
/// Independent of Gateway — manages its own AgentRuntime instances.
/// </summary>
public sealed class MobileTransport(
    IReadOnlyDictionary<string, AgentDefinition> agents,
    MobileSessionStore sessionStore,
    ILogger<MobileTransport> logger,
    CronService? cronService = null)
{
    private static readonly JsonSerializerOptions JsonOptions = new()
    {
        PropertyNamingPolicy = JsonNamingPolicy.SnakeCaseLower,
        DefaultIgnoreCondition = System.Text.Json.Serialization.JsonIgnoreCondition.WhenWritingNull,
    };

    private MobileConnection? _activeConnection;
    private readonly object _connectionLock = new();

    public MobileConnection? ActiveConnection => _activeConnection;

    public async Task AcceptAsync(WebSocket socket, CancellationToken ct)
    {
        // Read loop — parse frames and dispatch
        var buffer = new byte[16 * 1024]; // 16KB for larger frames
        var messageBuffer = new MemoryStream();

        try
        {
            while (socket.State == WebSocketState.Open && !ct.IsCancellationRequested)
            {
                messageBuffer.SetLength(0);
                WebSocketReceiveResult result;

                do
                {
                    result = await socket.ReceiveAsync(buffer, ct);
                    if (result.MessageType == WebSocketMessageType.Close) return;
                    messageBuffer.Write(buffer, 0, result.Count);
                } while (!result.EndOfMessage);

                var json = Encoding.UTF8.GetString(messageBuffer.GetBuffer(), 0, (int)messageBuffer.Length);
                await HandleFrameAsync(socket, json, ct);
            }
        }
        catch (WebSocketException) { /* client disconnected */ }
        catch (OperationCanceledException) { /* shutting down */ }
        finally
        {
            lock (_connectionLock)
            {
                if (_activeConnection?.Socket == socket)
                {
                    _ = _activeConnection.DisposeAsync();
                    _activeConnection = null;
                }
            }
        }
    }

    private async Task HandleFrameAsync(WebSocket socket, string json, CancellationToken ct)
    {
        Frame frame;
        try { frame = FrameParser.Parse(json); }
        catch (JsonException ex)
        {
            logger.LogWarning(ex, "Failed to parse frame");
            return;
        }

        switch (frame)
        {
            case RequestFrame req:
                await HandleRequestAsync(socket, req, ct);
                break;

            case ResponseFrame res:
                // Device command response — correlate with pending request
                _activeConnection?.TryCompleteRequest(res.Id, res);
                break;
        }
    }

    private async Task HandleRequestAsync(WebSocket socket, RequestFrame req, CancellationToken ct)
    {
        try
        {
            var response = req.Method switch
            {
                "connect" => await HandleConnectAsync(socket, req, ct),
                "ping" => ResponseFrame.Success(req.Id, JsonSerializer.SerializeToElement(new { })),
                "agents.list" => HandleAgentsList(req),
                "sessions.list" => await HandleSessionsListAsync(req, ct),
                "sessions.new" => await HandleSessionsNewAsync(req, ct),
                "sessions.switch" => await HandleSessionsSwitchAsync(req, ct),
                "sessions.rename" => await HandleSessionsRenameAsync(req, ct),
                "sessions.delete" => await HandleSessionsDeleteAsync(req, ct),
                "chat.send" => await HandleChatSendAsync(req, ct),
                "chat.cancel" => HandleChatCancel(req),
                _ => ResponseFrame.Failure(req.Id, "UNKNOWN", $"Unknown method: {req.Method}"),
            };

            if (_activeConnection is not null)
                await _activeConnection.SendFrameAsync(response, ct);
        }
        catch (Exception ex)
        {
            logger.LogError(ex, "Error handling RPC {Method}", req.Method);
            var error = ResponseFrame.Failure(req.Id, "UNKNOWN", ex.Message);
            if (_activeConnection is not null)
                await _activeConnection.SendFrameAsync(error, ct);
        }
    }

    private async Task<ResponseFrame> HandleConnectAsync(WebSocket socket, RequestFrame req, CancellationToken ct)
    {
        var agentName = req.Params.GetProperty("agent").GetString()!;
        var peerId = req.Params.GetProperty("peer").GetString()!;

        if (!agents.ContainsKey(agentName))
            return ResponseFrame.Failure(req.Id, "AGENT_NOT_FOUND", $"No agent named '{agentName}'");

        // Replace existing connection if any
        MobileConnection? old;
        lock (_connectionLock)
        {
            old = _activeConnection;
            _activeConnection = new MobileConnection(socket, peerId);
        }
        if (old is not null) await old.DisposeAsync();

        var conn = _activeConnection!;
        conn.CurrentAgent = agentName;
        conn.DeviceToken = req.Params.TryGetProperty("device_token", out var dt) ? dt.GetString() : null;

        if (req.Params.TryGetProperty("capabilities", out var caps))
        {
            foreach (var cap in caps.EnumerateArray())
            {
                if (cap.GetString() is string c) conn.Capabilities.Add(c);
            }
        }

        // Load or create session
        var sessions = await sessionStore.ListAsync(agentName, peerId, ct);
        MobileSession? currentSession;

        if (sessions.Count > 0)
        {
            currentSession = await sessionStore.LoadAsync(agentName, peerId, sessions[0].Id, ct);
        }
        else
        {
            currentSession = new MobileSession { Id = Guid.NewGuid().ToString("N"), Title = null };
            await sessionStore.SaveAsync(agentName, peerId, currentSession, ct);
        }

        conn.CurrentSessionId = currentSession!.Id;

        // Create AgentRuntime for current session
        var agentDef = agents[agentName];
        var runtime = CreateRuntime(agentDef, agentName, conn, currentSession.Messages);
        conn.SetRuntime(agentName, runtime);

        // Build response
        var agentList = agents.Select(kvp => new
        {
            name = kvp.Key,
            description = "", // filled in from config if available
            tools = kvp.Value.Tools.Select(t => t.Name).ToArray(),
        }).ToArray();

        var payload = JsonSerializer.SerializeToElement(new
        {
            agents = agentList,
            session = new
            {
                id = currentSession.Id,
                title = currentSession.Title,
                messages = currentSession.Messages.Count, // count only for connect
            },
            seq = 0,
        }, JsonOptions);

        return ResponseFrame.Success(req.Id, payload);
    }

    private ResponseFrame HandleAgentsList(RequestFrame req)
    {
        var agentList = agents.Select(kvp => new
        {
            name = kvp.Key,
            tools = kvp.Value.Tools.Select(t => t.Name).ToArray(),
        }).ToArray();

        return ResponseFrame.Success(req.Id, JsonSerializer.SerializeToElement(new { agents = agentList }, JsonOptions));
    }

    private async Task<ResponseFrame> HandleSessionsListAsync(RequestFrame req, CancellationToken ct)
    {
        var conn = _activeConnection;
        if (conn is null) return ResponseFrame.Failure(req.Id, "UNKNOWN", "Not connected");

        var agentName = req.Params.TryGetProperty("agent", out var a) ? a.GetString()! : conn.CurrentAgent!;
        var sessions = await sessionStore.ListAsync(agentName, conn.PeerId, ct);

        return ResponseFrame.Success(req.Id, JsonSerializer.SerializeToElement(new { sessions }, JsonOptions));
    }

    private async Task<ResponseFrame> HandleSessionsNewAsync(RequestFrame req, CancellationToken ct)
    {
        var conn = _activeConnection;
        if (conn is null) return ResponseFrame.Failure(req.Id, "UNKNOWN", "Not connected");

        var agentName = req.Params.TryGetProperty("agent", out var a) ? a.GetString()! : conn.CurrentAgent!;
        if (!agents.ContainsKey(agentName))
            return ResponseFrame.Failure(req.Id, "AGENT_NOT_FOUND", $"No agent named '{agentName}'");

        var session = new MobileSession { Id = Guid.NewGuid().ToString("N") };
        await sessionStore.SaveAsync(agentName, conn.PeerId, session, ct);

        // Replace runtime
        conn.RemoveRuntime(agentName);
        var agentDef = agents[agentName];
        var runtime = CreateRuntime(agentDef, agentName, conn, []);
        conn.SetRuntime(agentName, runtime);
        conn.CurrentAgent = agentName;
        conn.CurrentSessionId = session.Id;

        return ResponseFrame.Success(req.Id, JsonSerializer.SerializeToElement(new
        {
            session = new { id = session.Id, title = (string?)null, messages = 0 }
        }, JsonOptions));
    }

    private async Task<ResponseFrame> HandleSessionsSwitchAsync(RequestFrame req, CancellationToken ct)
    {
        var conn = _activeConnection;
        if (conn is null) return ResponseFrame.Failure(req.Id, "UNKNOWN", "Not connected");

        var sessionId = req.Params.GetProperty("session_id").GetString()!;
        var agentName = conn.CurrentAgent!;

        var session = await sessionStore.LoadAsync(agentName, conn.PeerId, sessionId, ct);
        if (session is null)
            return ResponseFrame.Failure(req.Id, "SESSION_NOT_FOUND", $"Session '{sessionId}' not found");

        // Replace runtime with loaded messages
        conn.RemoveRuntime(agentName);
        var agentDef = agents[agentName];
        var runtime = CreateRuntime(agentDef, agentName, conn, session.Messages);
        conn.SetRuntime(agentName, runtime);
        conn.CurrentSessionId = sessionId;

        return ResponseFrame.Success(req.Id, JsonSerializer.SerializeToElement(new
        {
            session = new { id = session.Id, title = session.Title, messages = session.Messages.Count }
        }, JsonOptions));
    }

    private async Task<ResponseFrame> HandleSessionsRenameAsync(RequestFrame req, CancellationToken ct)
    {
        var conn = _activeConnection;
        if (conn is null) return ResponseFrame.Failure(req.Id, "UNKNOWN", "Not connected");

        var sessionId = req.Params.GetProperty("session_id").GetString()!;
        var title = req.Params.GetProperty("title").GetString()!;

        await sessionStore.UpdateMetadataAsync(conn.CurrentAgent!, conn.PeerId, sessionId, title, ct);
        return ResponseFrame.Success(req.Id, JsonSerializer.SerializeToElement(new { }, JsonOptions));
    }

    private async Task<ResponseFrame> HandleSessionsDeleteAsync(RequestFrame req, CancellationToken ct)
    {
        var conn = _activeConnection;
        if (conn is null) return ResponseFrame.Failure(req.Id, "UNKNOWN", "Not connected");

        var sessionId = req.Params.GetProperty("session_id").GetString()!;
        await sessionStore.DeleteAsync(conn.CurrentAgent!, conn.PeerId, sessionId, ct);

        // If deleting the active session, create a new one
        if (conn.CurrentSessionId == sessionId)
        {
            var session = new MobileSession { Id = Guid.NewGuid().ToString("N") };
            await sessionStore.SaveAsync(conn.CurrentAgent!, conn.PeerId, session, ct);
            conn.RemoveRuntime(conn.CurrentAgent!);
            var agentDef = agents[conn.CurrentAgent!];
            var runtime = CreateRuntime(agentDef, conn.CurrentAgent!, conn, []);
            conn.SetRuntime(conn.CurrentAgent!, runtime);
            conn.CurrentSessionId = session.Id;
        }

        return ResponseFrame.Success(req.Id, JsonSerializer.SerializeToElement(new { }, JsonOptions));
    }

    private async Task<ResponseFrame> HandleChatSendAsync(RequestFrame req, CancellationToken ct)
    {
        var conn = _activeConnection;
        if (conn is null) return ResponseFrame.Failure(req.Id, "UNKNOWN", "Not connected");

        var text = req.Params.GetProperty("text").GetString()!;
        var agentName = conn.CurrentAgent!;
        var runtime = conn.GetRuntime(agentName);
        if (runtime is null)
            return ResponseFrame.Failure(req.Id, "UNKNOWN", "No active runtime");

        // Run agent turn in background — streaming events sent asynchronously
        _ = Task.Run(async () =>
        {
            try
            {
                AgentEventStream stream;
                if (runtime.IsRunning)
                {
                    runtime.FollowUp(new UserMessage { Text = text });
                    return; // Events will come from existing stream
                }

                stream = runtime.PromptAsync(text);
                await StreamEventsToClientAsync(conn, agentName, stream, ct);
            }
            catch (Exception ex)
            {
                logger.LogError(ex, "Error during chat.send for agent {Agent}", agentName);
            }
        }, ct);

        // Ack that streaming has started — events follow asynchronously
        return ResponseFrame.Success(req.Id, JsonSerializer.SerializeToElement(new { status = "streaming" }, JsonOptions));
    }

    private ResponseFrame HandleChatCancel(RequestFrame req)
    {
        var conn = _activeConnection;
        if (conn is null) return ResponseFrame.Failure(req.Id, "UNKNOWN", "Not connected");

        var runtime = conn.GetRuntime(conn.CurrentAgent!);
        runtime?.Abort();

        return ResponseFrame.Success(req.Id, JsonSerializer.SerializeToElement(new { }, JsonOptions));
    }

    private async Task StreamEventsToClientAsync(MobileConnection conn, string agentName, AgentEventStream stream, CancellationToken ct)
    {
        await foreach (var evt in stream.WithCancellation(ct))
        {
            switch (evt)
            {
                case MessageStreamEvent { Inner: TextDeltaEvent textDelta }:
                    await conn.SendEventAsync("text.delta",
                        JsonSerializer.SerializeToElement(new { delta = textDelta.Delta }, JsonOptions), ct);
                    break;

                case MessageStreamEvent { Inner: TextEndEvent }:
                    await conn.SendEventAsync("text.end",
                        JsonSerializer.SerializeToElement(new { }, JsonOptions), ct);
                    break;

                case MessageStreamEvent { Inner: ThinkingDeltaEvent thinkingDelta }:
                    await conn.SendEventAsync("thinking.delta",
                        JsonSerializer.SerializeToElement(new { delta = thinkingDelta.Delta }, JsonOptions), ct);
                    break;

                case MessageStreamEvent { Inner: ThinkingEndEvent }:
                    await conn.SendEventAsync("thinking.end",
                        JsonSerializer.SerializeToElement(new { }, JsonOptions), ct);
                    break;

                case ToolStartEvent toolStart:
                    await conn.SendEventAsync("tool.start",
                        JsonSerializer.SerializeToElement(new { tool = toolStart.ToolName, arguments = toolStart.Arguments }, JsonOptions), ct);
                    break;

                case ToolEndEvent toolEnd:
                    var resultText = toolEnd.Result.Content
                        .OfType<Achates.Providers.Completions.CompletionTextContent>()
                        .FirstOrDefault()?.Text ?? "";
                    await conn.SendEventAsync("tool.end",
                        JsonSerializer.SerializeToElement(new { result = resultText, error = toolEnd.IsError }, JsonOptions), ct);
                    break;

                case MessageEndEvent msgEnd when msgEnd.Message is AssistantMessage am:
                    await conn.SendEventAsync("message.end",
                        JsonSerializer.SerializeToElement(new
                        {
                            usage = new
                            {
                                input = am.Usage.Input,
                                output = am.Usage.Output,
                                cost = am.Usage.Cost.Input + am.Usage.Cost.Output,
                            }
                        }, JsonOptions), ct);
                    break;

                case AgentEndEvent:
                    await conn.SendEventAsync("done",
                        JsonSerializer.SerializeToElement(new { }, JsonOptions), ct);

                    // Persist session
                    var runtime = conn.GetRuntime(agentName);
                    if (runtime is not null && conn.CurrentSessionId is not null)
                    {
                        var session = await sessionStore.LoadAsync(agentName, conn.PeerId, conn.CurrentSessionId, ct)
                            ?? new MobileSession { Id = conn.CurrentSessionId };
                        session.Messages = [.. runtime.Messages];
                        await sessionStore.SaveAsync(agentName, conn.PeerId, session, ct);
                    }
                    break;
            }
        }
    }

    private AgentRuntime CreateRuntime(AgentDefinition agentDef, string agentName, MobileConnection conn, IReadOnlyList<AgentMessage>? messages)
    {
        // Build per-session tools (same pattern as Gateway.BuildSessionTools)
        var tools = new List<AgentTool>(agentDef.Tools);
        // MemoryTool, TodoTool, CostTool are added here like Gateway does
        // For now, just use the shared tools from agentDef

        return new AgentRuntime(new AgentOptions
        {
            Model = agentDef.Model,
            SystemPrompt = agentDef.SystemPrompt,
            Tools = tools,
            Messages = messages is { Count: > 0 } ? [.. messages] : null,
            CompletionOptions = agentDef.CompletionOptions,
        });
    }
}
```

```csharp
using System.Text.Json;

namespace Achates.Server.Mobile;

/// <summary>
/// Singleton bridge between server-side tools and the mobile client.
/// Tools call this to invoke device capabilities (location, camera).
/// </summary>
public sealed class DeviceCommandBridge
{
    private static readonly JsonSerializerOptions JsonOptions = new()
    {
        PropertyNamingPolicy = JsonNamingPolicy.SnakeCaseLower,
    };

    private MobileTransport? _transport;

    public void SetTransport(MobileTransport transport) => _transport = transport;

    public bool IsAvailable(string capability)
    {
        var conn = _transport?.ActiveConnection;
        return conn is not null && conn.Capabilities.Contains(capability);
    }

    public async Task<JsonElement?> InvokeAsync(string method, JsonElement? parameters = null, TimeSpan? timeout = null, CancellationToken ct = default)
    {
        var conn = _transport?.ActiveConnection;
        if (conn is null) return null;

        var response = await conn.SendRequestAsync(method, parameters, timeout, ct);
        if (response is null || !response.Ok) return null;
        return response.Payload;
    }
}
```

```csharp
private async Task TryNameSessionAsync(MobileConnection conn, string agentName, CancellationToken ct)
{
    if (conn.CurrentSessionId is null) return;
    var session = await sessionStore.LoadAsync(agentName, conn.PeerId, conn.CurrentSessionId, ct);
    if (session is null || session.Title is not null) return; // Already named
    if (session.Messages.Count < 2) return; // Need at least user + assistant

    var firstUser = session.Messages.OfType<UserMessage>().FirstOrDefault()?.Text;
    var firstAssistant = session.Messages.OfType<AssistantMessage>().FirstOrDefault();
    if (firstUser is null || firstAssistant is null) return;

    var firstAssistantText = string.Join("", firstAssistant.Content
        .OfType<Achates.Providers.Completions.CompletionTextContent>()
        .Select(c => c.Text));

    try
    {
        var agentDef = agents[agentName];
        var namingRuntime = new AgentRuntime(new AgentOptions
        {
            Model = agentDef.Model,
            SystemPrompt = "Generate a short title (3-6 words) for this conversation. Respond with ONLY the title, no quotes, no punctuation at the end.",
        });

        var stream = namingRuntime.PromptAsync(prompt);
        var result = await stream.ResultAsync;

        var title = result.OfType<AssistantMessage>().FirstOrDefault()?.Content

            await conn.SendEventAsync("session.renamed",
                JsonSerializer.SerializeToElement(new { session_id = conn.CurrentSessionId, title }, JsonOptions), ct);

    catch (Exception ex)
```

```swift
// ios/Achates/Models/Frame.swift
import Foundation

enum FrameType: String, Codable {
    case req, res, evt
}

struct RequestFrame: Codable {
    let type: String
    let id: String
    let method: String
    let params: [String: AnyCodable]?

    init(method: String, params: [String: AnyCodable]? = nil) {
        self.type = "req"
        self.id = UUID().uuidString.lowercased().replacingOccurrences(of: "-", with: "")
        self.method = method
        self.params = params
    }
}

struct ResponseFrame: Codable {
    let type: String
    let id: String
    let ok: Bool
    let payload: AnyCodable?
    let error: FrameError?
}

struct FrameError: Codable {
    let code: String
    let message: String
}

struct EventFrame: Codable {
    let type: String
    let event: String
    let payload: AnyCodable?
    let seq: Int
}

// Generic JSON value wrapper for Codable
enum AnyCodable: Codable {
    case string(String)
    case int(Int)
    case double(Double)
    case bool(Bool)
    case dictionary([String: AnyCodable])
    case array([AnyCodable])
    case null

    init(from decoder: Decoder) throws {
        let container = try decoder.singleValueContainer()
        if container.decodeNil() { self = .null; return }
        if let v = try? container.decode(Bool.self) { self = .bool(v); return }
        if let v = try? container.decode(Int.self) { self = .int(v); return }
        if let v = try? container.decode(Double.self) { self = .double(v); return }
        if let v = try? container.decode(String.self) { self = .string(v); return }
        if let v = try? container.decode([String: AnyCodable].self) { self = .dictionary(v); return }
        if let v = try? container.decode([AnyCodable].self) { self = .array(v); return }
        throw DecodingError.dataCorruptedError(in: container, debugDescription: "Unsupported type")
    }

    func encode(to encoder: Encoder) throws {
        var container = encoder.singleValueContainer()
        switch self {
        case .string(let v): try container.encode(v)
        case .int(let v): try container.encode(v)
        case .double(let v): try container.encode(v)
        case .bool(let v): try container.encode(v)
        case .dictionary(let v): try container.encode(v)
        case .array(let v): try container.encode(v)
        case .null: try container.encodeNil()
        }
    }

    var stringValue: String? {
        if case .string(let v) = self { return v }
        return nil
    }

    var intValue: Int? {
        if case .int(let v) = self { return v }
        return nil
    }

    var dictionaryValue: [String: AnyCodable]? {
```

```swift
// ios/Achates/Connection/WebSocketClient.swift
import Foundation

final class WebSocketClient {
    private let url: URL
    private weak var state: AppState?
    private var webSocketTask: URLSessionWebSocketTask?
    private let pendingLock = NSLock()
    private var pendingRequests: [String: CheckedContinuation<ResponseFrame?, Never>] = [:]
    private let encoder = JSONEncoder()
    private let decoder = JSONDecoder()

    init(url: URL, state: AppState) {
        self.url = url
        self.state = state
        encoder.keyEncodingStrategy = .convertToSnakeCase
        decoder.keyDecodingStrategy = .convertFromSnakeCase
    }

    func connect() {
        let wsURL = url.appendingPathComponent("ws/v2")
        let session = URLSession(configuration: .default)
        webSocketTask = session.webSocketTask(with: wsURL)
        webSocketTask?.resume()
        receiveLoop()
    }

    func disconnect() {
        webSocketTask?.cancel(with: .normalClosure, reason: nil)
        webSocketTask = nil
    }

    func sendRequest(_ method: String, params: [String: AnyCodable]? = nil) async -> ResponseFrame? {
        let frame = RequestFrame(method: method, params: params)
        guard let data = try? encoder.encode(frame),
              let json = String(data: data, encoding: .utf8) else { return nil }

        return await withCheckedContinuation { continuation in
            pendingLock.lock()
            pendingRequests[frame.id] = continuation
            pendingLock.unlock()

            webSocketTask?.send(.string(json)) { [weak self] error in
                if error != nil {
                    self?.pendingLock.lock()
                    self?.pendingRequests.removeValue(forKey: frame.id)
                    self?.pendingLock.unlock()
                    continuation.resume(returning: nil)
                }
            }
        }
    }

    func sendConnect(agent: String, peer: String, capabilities: [String] = [], lastSeq: Int? = nil) async -> ResponseFrame? {
        var params: [String: AnyCodable] = [
            "agent": .string(agent),
            "peer": .string(peer),
            "capabilities": .array(capabilities.map { .string($0) }),
        ]
        if let lastSeq { params["last_seq"] = .int(lastSeq) }
        return await sendRequest("connect", params: params)
    }

    private func receiveLoop() {
        webSocketTask?.receive { [weak self] result in
            guard let self else { return }
            switch result {
            case .success(.string(let text)):
                self.handleMessage(text)
                self.receiveLoop() // Continue receiving
            case .success(.data(let data)):
                if let text = String(data: data, encoding: .utf8) {
                    self.handleMessage(text)
                }
                self.receiveLoop()
            case .failure:
                Task { @MainActor in
                    self.state?.connectionStatus = .disconnected
                }
            default:
                self.receiveLoop()
            }
        }
    }

    private func handleMessage(_ text: String) {
        guard let data = text.data(using: .utf8) else { return }

        // Determine frame type
        struct TypePeek: Decodable { let type: String }
        guard let peek = try? decoder.decode(TypePeek.self, from: data) else { return }

        switch peek.type {
        case "res":
            if let response = try? decoder.decode(ResponseFrame.self, from: data) {
                pendingLock.lock()
                let continuation = pendingRequests.removeValue(forKey: response.id)
                pendingLock.unlock()
                continuation?.resume(returning: response)
            }

        case "evt":
            if let event = try? decoder.decode(EventFrame.self, from: data) {
                Task { @MainActor in
                    self.handleEvent(event)
                }
            }

        case "req":
            // Server-to-client request (device commands)
            if let request = try? decoder.decode(RequestFrame.self, from: data) {
                Task {
                    await self.handleServerRequest(request)
                }
            }

        default:
            break
        }
    }

    @MainActor
    private func handleEvent(_ event: EventFrame) {
        guard let state else { return }

        switch event.event {
        case "text.delta":
            let delta = event.payload?.dictionaryValue?["delta"]?.stringValue ?? ""
            if let id = state.streamingMessageId,
               let idx = state.messages.firstIndex(where: { $0.id == id }) {
                state.messages[idx].text += delta
            } else {
                let msg = ChatMessage(role: .assistant, text: delta)
                state.messages.append(msg)
                state.streamingMessageId = msg.id
                state.isStreaming = true
            }

        case "text.end":
            break // text is already accumulated

        case "thinking.delta":
            let delta = event.payload?.dictionaryValue?["delta"]?.stringValue ?? ""
            if let id = state.streamingMessageId,
               let idx = state.messages.firstIndex(where: { $0.id == id }) {
                state.messages[idx].thinkingText = (state.messages[idx].thinkingText ?? "") + delta
            }

        case "thinking.end":
            break

        case "tool.start":
            let toolName = event.payload?.dictionaryValue?["tool"]?.stringValue ?? "unknown"
            let args = event.payload?.dictionaryValue?["arguments"]?.dictionaryValue
            let argsStr = args.map { dict -> String in
                (try? String(data: JSONEncoder().encode(dict), encoding: .utf8)) ?? "{}"
            } ?? "{}"

            let toolCall = ToolCall(id: UUID().uuidString, name: toolName, arguments: argsStr)
            if let id = state.streamingMessageId,
               let idx = state.messages.firstIndex(where: { $0.id == id }) {
                state.messages[idx].toolCalls.append(toolCall)
            }

        case "tool.end":
            let result = event.payload?.dictionaryValue?["result"]?.stringValue ?? ""
            let isError = { if case .bool(let v) = event.payload?.dictionaryValue?["error"] { return v }; return false }()
            if let id = state.streamingMessageId,
               let idx = state.messages.firstIndex(where: { $0.id == id }),
               !state.messages[idx].toolCalls.isEmpty {
                let toolIdx = state.messages[idx].toolCalls.count - 1
                state.messages[idx].toolCalls[toolIdx].result = result
                state.messages[idx].toolCalls[toolIdx].isError = isError
                state.messages[idx].toolCalls[toolIdx].isComplete = true
            }

        case "message.end":
            break // Usage stats — could display later

        case "done":
            state.isStreaming = false
            state.streamingMessageId = nil

        case "session.renamed":
            let title = event.payload?.dictionaryValue?["title"]?.stringValue
            let sessionId = event.payload?.dictionaryValue?["session_id"]?.stringValue
            if let title, let sessionId, state.currentSession?.id == sessionId {
                // Update would need mutable session — handle in implementation
            }

        default:
            break
        }
    }

    private func handleServerRequest(_ request: RequestFrame) async {
        // Device command handling — will be implemented in Services/DeviceCommandRouter.swift
        let response = ResponseFrame(
            type: "res",
            id: request.id,
            ok: false,
            payload: nil,
            error: FrameError(code: "DEVICE_UNAVAILABLE", message: "Not implemented yet")
        )
        guard let data = try? encoder.encode(response),
              let json = String(data: data, encoding: .utf8) else { return }
        try? await webSocketTask?.send(.string(json))
    }
}
```

```swift
                  let id = dict["id"]?.stringValue else { return nil }
            return Session(
```

```swift
import AVFoundation

final class CameraService {
    func capturePhoto(facing: String = "back") async throws -> Data {
        // Use AVCaptureSession for programmatic capture
        let session = AVCaptureSession()
        session.sessionPreset = .photo

        let position: AVCaptureDevice.Position = facing == "front" ? .front : .back
        guard let device = AVCaptureDevice.default(.builtInWideAngleCamera, for: .video, position: position) else {
            throw CameraError.noCamera
        }

        let input = try AVCaptureDeviceInput(device: device)
        session.addInput(input)

        let output = AVCapturePhotoOutput()
        session.addOutput(output)

        session.startRunning()
        defer { session.stopRunning() }

        let delegate = PhotoCaptureDelegate()
        let settings = AVCapturePhotoSettings(format: [AVVideoCodecKey: AVVideoCodecType.jpeg])
        output.capturePhoto(with: settings, delegate: delegate)

        let imageData = try await delegate.result()

        // Compress to max ~500KB
        guard let image = UIImage(data: imageData) else { throw CameraError.captureFailure }
        var quality: CGFloat = 0.8
        var compressed = image.jpegData(compressionQuality: quality) ?? imageData
        while compressed.count > 500_000 && quality > 0.1 {
            quality -= 0.1
            compressed = image.jpegData(compressionQuality: quality) ?? compressed
        }

        return compressed
    }

    enum CameraError: Error {
        case noCamera
        case captureFailure
    }
}

private final class PhotoCaptureDelegate: NSObject, AVCapturePhotoCaptureDelegate {
    private var continuation: CheckedContinuation<Data, Error>?

    func result() async throws -> Data {
        try await withCheckedThrowingContinuation { continuation in

    }

    func photoOutput(_ output: AVCapturePhotoOutput, didFinishProcessingPhoto photo: AVCapturePhoto, error: Error?) {
        if let error {
            continuation?.resume(throwing: error)
        } else if let data = photo.fileDataRepresentation() {
            continuation?.resume(returning: data)
```

```swift
// ios/Achates/Services/DeviceCommandRouter.swift
import Foundation

final class DeviceCommandRouter {
    let locationService = LocationService()
    let cameraService = CameraService()

    func handle(_ request: RequestFrame) async -> ResponseFrame {
        switch request.method {
        case "device.location":
            return await handleLocation(request)
        case "device.camera":
            return await handleCamera(request)
        default:
            return ResponseFrame(
                type: "res", id: request.id, ok: false, payload: nil,
                error: FrameError(code: "UNKNOWN", message: "Unknown device command: \(request.method)")
            )
        }
    }

    private func handleLocation(_ request: RequestFrame) async -> ResponseFrame {
        do {
            let loc = try await locationService.getCurrentLocation()
            let payload: [String: AnyCodable] = [
                "lat": .double(loc.lat),
                "lon": .double(loc.lon),
                "accuracy": .double(loc.accuracy),
                "timestamp": .string(ISO8601DateFormatter().string(from: Date()))
            ]
            return ResponseFrame(
                type: "res", id: request.id, ok: true,
                payload: .dictionary(payload), error: nil
            )
        } catch {
            return ResponseFrame(
                type: "res", id: request.id, ok: false, payload: nil,
                error: FrameError(code: "DEVICE_TIMEOUT", message: error.localizedDescription)
            )
        }
    }

    private func handleCamera(_ request: RequestFrame) async -> ResponseFrame {
        do {
            let facing = request.params?["facing"]?.stringValue ?? "back"
            let imageData = try await cameraService.capturePhoto(facing: facing)
            let base64 = imageData.base64EncodedString()
            let payload: [String: AnyCodable] = [
                "base64": .string(base64),
                "metadata": .dictionary([
                    "size": .int(imageData.count),
                    "format": .string("jpeg"),
                ])
            ]
            return ResponseFrame(
                type: "res", id: request.id, ok: true,
                payload: .dictionary(payload), error: nil
            )
        } catch {
            return ResponseFrame(
                type: "res", id: request.id, ok: false, payload: nil,
                error: FrameError(code: "DEVICE_TIMEOUT", message: error.localizedDescription)
            )
        }
    }
}
```

```swift
import Speech
import AVFoundation

@Observable
final class SpeechService {
    var transcription = ""
    var isRecording = false
    var isAvailable = false

    private let recognizer = SFSpeechRecognizer()
    private var recognitionRequest: SFSpeechAudioBufferRecognitionRequest?
    private var recognitionTask: SFSpeechRecognitionTask?
    private let audioEngine = AVAudioEngine()

    init() {
        isAvailable = recognizer?.isAvailable ?? false
    }

    func requestPermission() async -> Bool {
        await withCheckedContinuation { continuation in
            SFSpeechRecognizer.requestAuthorization { status in
                continuation.resume(returning: status == .authorized)
            }
        }
    }

    func startRecording() throws {
        recognitionTask?.cancel()
        recognitionTask = nil

        let audioSession = AVAudioSession.sharedInstance()
        try audioSession.setCategory(.record, mode: .measurement, options: .duckOthers)
        try audioSession.setActive(true, options: .notifyOthersOnDeactivation)

        recognitionRequest = SFSpeechAudioBufferRecognitionRequest()
        guard let request = recognitionRequest else { return }
        request.shouldReportPartialResults = true
        request.requiresOnDeviceRecognition = true

        recognitionTask = recognizer?.recognitionTask(with: request) { [weak self] result, error in
            guard let self else { return }
            if let result {
                Task { @MainActor in
                    self.transcription = result.bestTranscription.formattedString
                }
            }
            if error != nil || result?.isFinal == true {
                Task { @MainActor in
                    self.stopRecording()
                }
            }
        }

        let inputNode = audioEngine.inputNode
        let format = inputNode.outputFormat(forBus: 0)
        inputNode.installTap(onBus: 0, bufferSize: 1024, format: format) { buffer, _ in
            request.append(buffer)
        }

        audioEngine.prepare()
        try audioEngine.start()
        isRecording = true
    }

    func stopRecording() {
        audioEngine.stop()
        audioEngine.inputNode.removeTap(onBus: 0)
        recognitionRequest?.endAudio()
        recognitionRequest = nil
        recognitionTask = nil
        isRecording = false
    }
}
```

```csharp
using System.Collections.Concurrent;
using System.IdentityModel.Tokens.Jwt;
using System.Net.Http.Headers;
using System.Security.Cryptography;
using System.Text;
using System.Text.Json;
using Achates.Configuration;
using Microsoft.Extensions.Logging;
using Microsoft.IdentityModel.Tokens;

namespace Achates.Server.Mobile;

public sealed class PushNotificationService(ApnsConfig config, HttpClient httpClient, ILogger<PushNotificationService> logger)
{
    private readonly ConcurrentDictionary<string, string> _deviceTokens = new(); // peerId -> token

    public void RegisterToken(string peerId, string token) => _deviceTokens[peerId] = token;

    public async Task SendAsync(string peerId, string agentName, string? sessionId, string preview, CancellationToken ct = default)
    {
        if (!_deviceTokens.TryGetValue(peerId, out var token))
        {
            logger.LogDebug("No device token for peer {PeerId}", peerId);
            return;
        }

        var host = config.Environment == "production"
            ? "https://api.push.apple.com"
            : "https://api.development.push.apple.com";

        var payload = new
        {
            aps = new
            {
                alert = new { title = agentName, body = preview },
                badge = 1,
                sound = "default",
            },
            agent = agentName,
            session_id = sessionId,
        };

        var json = JsonSerializer.Serialize(payload);
        var request = new HttpRequestMessage(HttpMethod.Post, $"{host}/3/device/{token}")
        {
            Content = new StringContent(json, Encoding.UTF8, "application/json"),
            Version = new Version(2, 0),
        };

        request.Headers.Authorization = new AuthenticationHeaderValue("bearer", GenerateJwt());
        request.Headers.TryAddWithoutValidation("apns-topic", config.BundleId);
        request.Headers.TryAddWithoutValidation("apns-push-type", "alert");

        try
        {
            var response = await httpClient.SendAsync(request, ct);
            if (!response.IsSuccessStatusCode)
                logger.LogWarning("APNs push failed: {Status}", response.StatusCode);
        }
        catch (Exception ex)
        {
            logger.LogError(ex, "APNs push error for peer {PeerId}", peerId);
        }
    }

    private string GenerateJwt()
    {
        var keyPath = config.KeyFile!.Replace("~", Environment.GetFolderPath(Environment.SpecialFolder.UserProfile));
        var keyData = File.ReadAllText(keyPath);
        var key = ECDsa.Create();
        key.ImportFromPem(keyData);

        var signingCredentials = new SigningCredentials(new ECDsaSecurityKey(key), SecurityAlgorithms.EcdsaSha256);
        var now = DateTime.UtcNow;

        var token = new JwtSecurityToken(
            issuer: config.TeamId,
            claims: [],
            notBefore: now,
            expires: now.AddMinutes(55),
            signingCredentials: signingCredentials
        );
        token.Header["kid"] = config.KeyId;

        return new JwtSecurityTokenHandler().WriteToken(token);
    }
}
```

```swift
private var reconnectAttempt = 0
private var maxReconnectDelay: TimeInterval = 30
private var lastBackgroundTime: Date?

func handleAppDidEnterBackground() {
    lastBackgroundTime = Date()
}

func handleAppWillEnterForeground() {
    guard let backgroundTime = lastBackgroundTime else { return }
    let elapsed = Date().timeIntervalSince(backgroundTime)
    if elapsed >= 3 {
        // Health check
        Task {
            let response = await sendRequest("ping")
            if response == nil {
                reconnect()
            }
        }
    }
    lastBackgroundTime = nil
}

private func reconnect() {
    disconnect()
    let delay = min(pow(2, Double(reconnectAttempt)) * 0.5, maxReconnectDelay)
    reconnectAttempt += 1

    Task {
        try? await Task.sleep(for: .seconds(delay))
        connect()
        // Re-send connect with last_seq for event replay
        if let state, let agent = state.currentAgent {
            let response = await sendConnect(
                agent: agent.name,
                peer: UIDevice.current.identifierForVendor?.uuidString ?? "ios",
                capabilities: ["location", "camera"],
                lastSeq: lastSeq
            )
            if response?.ok == true {
                await MainActor.run { state.connectionStatus = .connected }
                reconnectAttempt = 0
            }
        }
    }
}
```
